CREATE-Scripts:
CREATE TABLE users(
	user VARCHAR(255) PRIMARY KEY,
	password VARCHAR(255),
	name VARCHAR(255),
	token VARCHAR(255) UNIQUE
);

CREATE TABLE playlists(
	playlistId INT AUTO_INCREMENT PRIMARY KEY, 
	name VARCHAR(255) UNIQUE,
	owner BIT
);

CREATE TABLE tracks(
	trackId INT AUTO_INCREMENT PRIMARY KEY,
	title VARCHAR(255),
	performer VARCHAR(255),
	duration INT,
	album VARCHAR(255)  DEFAULT 'undefined',
	playcount INT DEFAULT 0,
	publicationDate VARCHAR(255) DEFAULT 'undefined',
	description VARCHAR(255) DEFAULT 'undefined',
	offlineAvailable BIT DEFAULT 0,
	UNIQUE (title, performer)
);

CREATE TABLE playlistTrack(	
	playlistId INT,
	trackId INT,
	PRIMARY KEY(playlistId, trackId),
	FOREIGN KEY(playlistId) REFERENCES playlists(playlistId),
	FOREIGN KEY(trackId) REFERENCES tracks(trackId)
);

CREATE TABLE playlistUser(
	playlistId INT,
	user VARCHAR(255),
	PRIMARY KEY(playlistId, user),
	FOREIGN KEY(playlistId) REFERENCES playlists(playlistId),
	FOREIGN KEY(user) REFERENCES users(user)
);
